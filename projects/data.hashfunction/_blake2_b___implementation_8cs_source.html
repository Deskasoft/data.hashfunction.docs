190 var tempH = _h;

192 var tempCounter = _counter;

194
195 // Process _delayedInputBuffer
196 if (_delayedInputBuffer != null)
197 {
198 tempCounter += new UInt128(128);
199
200 Compress(
201 ref tempA, ref tempB, ref tempC, ref tempD, ref tempE, ref tempF, ref tempG, ref tempH,
202 tempCounter,
203 _delayedInputBuffer, 0,
204 false);
205 }
206
207 // Delay the last 128 bytes of input
208 {
209 _delayedInputBuffer = new byte[128];
210 Array.Copy(dataArray, dataOffset + dataCount - 128, _delayedInputBuffer, 0, 128);

245 {
246 var remainder = FinalizeInputBuffer;
247 var remainderCount = (remainder?.Length).GetValueOrDefault();
248
249 var tempA = _a;
250 var tempB = _b;
251 var tempC = _c;
252 var tempD = _d;
253 var tempE = _e;
254 var tempF = _f;
255 var tempG = _g;
256 var tempH = _h;
257
258 var tempCounter = _counter;

260
261 if (_delayedInputBuffer != null)
262 {
263 cancellationToken.ThrowIfCancellationRequested();
264
265
266 tempCounter += new UInt128(128);
267
268 Compress(
269 ref tempA, ref tempB, ref tempC, ref tempD, ref tempE, ref tempF, ref tempG, ref tempH,
270 tempCounter,
271 _delayedInputBuffer, 0,
272 remainderCount == 0);
273 }
274
275
276 if (remainderCount > 0 || _delayedInputBuffer == null)
277 {
278 cancellationToken.ThrowIfCancellationRequested();

280 var finalBuffer = new byte[128];
281
282 if (remainderCount > 0)
283 Array.Copy(remainder, 0, finalBuffer, 0, remainderCount);
284
285
286 tempCounter += new UInt128((UInt64) remainderCount);
287
288 Compress(
289 ref tempA, ref tempB, ref tempC, ref tempD, ref tempE, ref tempF, ref tempG, ref tempH,
290 tempCounter,
291 finalBuffer, 0,
292 true);
293 }

295
296 var hashValueBytes = BitConverter.GetBytes(tempA)
297 .Concat(BitConverter.GetBytes(tempB))
298 .Concat(BitConverter.GetBytes(tempC))
299 .Concat(BitConverter.GetBytes(tempD))
300 .Concat(BitConverter.GetBytes(tempE))
301 .Concat(BitConverter.GetBytes(tempF))
302 .Concat(BitConverter.GetBytes(tempG))

304 .Take(_hashSizeInBits / 8)
