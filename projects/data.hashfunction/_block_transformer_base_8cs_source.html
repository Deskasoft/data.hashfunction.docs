237 protected void TransformBytesInternal(ArraySegment<byte> data, CancellationToken cancellationToken)
238 {
239 var dataArray = data.Array;
240 var dataCurrentOffset = data.Offset;
241 var dataRemainingCount = data.Count;
242
243 var totalBytesToProcess = (_inputBuffer?.Length).GetValueOrDefault() + dataRemainingCount;
244 var processingRemainder = totalBytesToProcess % _inputBlockSize;
245
246 // Determine how many bytes will inevitably be rolled over into the next input buffer and prepare that buffer.
247 byte[] nextInputBuffer = null;
248 {
249 if (processingRemainder > 0)
250 {
251 nextInputBuffer = new byte[processingRemainder];
252
253 if (dataRemainingCount >= processingRemainder)
254 {
255 // All of the data can be fetched from newest data
256 var copyFromOffset = dataCurrentOffset + dataRemainingCount - processingRemainder;
257
258 Array.Copy(dataArray, copyFromOffset, nextInputBuffer, 0, processingRemainder);
259
260 dataRemainingCount -= processingRemainder;
261
262 }
263 else
264 {
265 var bytesRolledOver = 0;
266
267 if (_inputBuffer != null)
268 {
269 Array.Copy(_inputBuffer, nextInputBuffer, _inputBuffer.Length);
270
271 bytesRolledOver = _inputBuffer.Length;
272 }
273
274 Array.Copy(dataArray, 0, nextInputBuffer, bytesRolledOver, processingRemainder - bytesRolledOver);
275
276 dataRemainingCount = 0;
277 }
278 }
279 }
280
281 // Process the full groups available
282 if (dataRemainingCount > 0)
283 {
284 try
285 {
286 if (_inputBuffer != null)
287 {
288 var overrideInputBuffer = new byte[_inputBlockSize];
289 var dataBytesToRead = overrideInputBuffer.Length - _inputBuffer.Length;
290
291 Array.Copy(_inputBuffer, overrideInputBuffer, _inputBuffer.Length);
292 Array.Copy(dataArray, dataCurrentOffset, overrideInputBuffer, _inputBuffer.Length, dataBytesToRead);
293
294 dataCurrentOffset += dataBytesToRead;
295 dataRemainingCount -= dataBytesToRead;
296
297
298 TransformByteGroupsInternal(new ArraySegment<byte>(overrideInputBuffer, 0, overrideInputBuffer.Length));
299 }
300
301 while (dataRemainingCount > 0)
302 {
303 cancellationToken.ThrowIfCancellationRequested();
304
305 var bytesToProcess = Math.Min(dataRemainingCount, _cancellationBatchSize);
306
307 TransformByteGroupsInternal(new ArraySegment<byte>(data.Array, dataCurrentOffset, bytesToProcess));
308
309 dataCurrentOffset += bytesToProcess;
310 dataRemainingCount -= bytesToProcess;
311 }
312
313 Debug.Assert(dataRemainingCount == 0);
314 }
315 catch (TaskCanceledException)
316 {
317 MarkSelfCorrupted();
318
319 throw;
320 }
321 }
322
323 // Update input buffer
324 _inputBuffer = nextInputBuffer;
325 }
