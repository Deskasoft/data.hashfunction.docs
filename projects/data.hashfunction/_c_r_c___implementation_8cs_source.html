112 protected override void TransformByteGroupsInternal(ArraySegment<byte> data)
113 {
114 var dataArray = data.Array;
115 var dataOffset = data.Offset;
116 var endOffset = dataOffset + data.Count;
117
118 var tempHashValue = _hashValue;
119
120 var tempHashSizeInBits = _hashSizeInBits;
121 var tempReflectIn = _reflectIn;
122 var tempCrcTable = _crcTable;
123 var tempMostSignificantShift = _mostSignificantShift;
124
125 for (var currentOffset = dataOffset; currentOffset < endOffset; ++currentOffset)
126 {
127 if (tempHashSizeInBits >= 8)
128 {
129 // Process per byte, treating hash differently based on input endianness
130 if (tempReflectIn)
131 tempHashValue = (tempHashValue >> 8) ^ tempCrcTable[(byte) tempHashValue ^ dataArray[currentOffset]];
132 else
133 tempHashValue = (tempHashValue << 8) ^ tempCrcTable[((byte) (tempHashValue >> tempMostSignificantShift)) ^ dataArray[currentOffset]];
134
135 } else {
136 // Process per bit, treating hash differently based on input endianness
137 for (var currentBit = 0; currentBit < 8; ++currentBit)
138 {
139 if (tempReflectIn)
140 tempHashValue = (tempHashValue >> 1) ^ tempCrcTable[(byte) (tempHashValue & 1) ^ ((byte) (dataArray[currentOffset] >> currentBit) & 1)];
141 else
142 tempHashValue = (tempHashValue << 1) ^ tempCrcTable[(byte) ((tempHashValue >> tempMostSignificantShift) & 1) ^ ((byte) (dataArray[currentOffset] >> (7 - currentBit)) & 1)];
143 }
144
145 }
146 }
147
148 _hashValue = tempHashValue;

190 private static IReadOnlyList<UInt64> GetDataDivisionTableInternal((int, UInt64, bool) cacheKey)
191 {
192 var hashSizeInBits = cacheKey.Item1;
193 var polynomial = cacheKey.Item2;
194 var reflectIn = cacheKey.Item3;
195
196
197 var perBitCount = 8;
198
199 if (hashSizeInBits < 8)
200 perBitCount = 1;
201
202
203 var crcTable = new UInt64[1 << perBitCount];
204 var mostSignificantBit = 1UL << (hashSizeInBits - 1);
205
206
207 for (uint x = 0; x < crcTable.Length; ++x)
208 {
209 UInt64 curValue = x;
210
211 if (perBitCount > 1 && reflectIn)
212 curValue = ReflectBits(curValue, perBitCount);
213
214
215 curValue <<= (hashSizeInBits - perBitCount);
216
217
218 for (int y = 0; y < perBitCount; ++y)
219 {
220 if ((curValue & mostSignificantBit) > 0UL)
221 curValue = (curValue << 1) ^ polynomial;
222 else
223 curValue <<= 1;
224 }
225
226
227 if (reflectIn)
228 curValue = ReflectBits(curValue, hashSizeInBits);
229
230
231 curValue &= (UInt64.MaxValue >> (64 - hashSizeInBits));
232
233 crcTable[x] = curValue;
234 }
235
236
237 return crcTable;
238 }
