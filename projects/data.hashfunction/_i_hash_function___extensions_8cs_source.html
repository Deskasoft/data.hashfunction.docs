464 public static IHashValue ComputeHash(this IHashFunction hashFunction, byte[] data, int desiredHashSize)
465 {
466 var hash = new BigInteger();
467 var desiredHashBytes = (desiredHashSize + 7) / 8;
468
469 var seededData = new byte[data.Length + 4];
470 Array.Copy(data, 0, seededData, 4, data.Length);
471
472 var hashesNeeded = (desiredHashSize + (hashFunction.HashSizeInBits - 1)) / hashFunction.HashSizeInBits;
473
474 // Compute as many hashes as needed
475 for (int x = 0; x < Math.Max(hashesNeeded, 1); ++x)
476 {
477 byte[] currentData;
478
479 if (x != 0)
480 {
481 Array.Copy(BitConverter.GetBytes(x), seededData, 4);
482 currentData = seededData;
483
484 }
485 else
486 {
487 // Use original data for first
488 currentData = data;
489 }
490
491
492 var elementHash = new BigInteger(
493 hashFunction.ComputeHash(currentData)
494 .Hash
495 .Concat(new[] { (byte) 0 })
496 .ToArray());
497
498 hash |= elementHash << (x * hashFunction.HashSizeInBits);
499 }
500
501
502 // XOr-fold the extra bits
503 if (hashesNeeded * hashFunction.HashSizeInBits != desiredHashSize)
504 {
505 var mask = ((new BigInteger(1) << desiredHashSize) - 1);
506
507 hash = (hash ^ (hash >> desiredHashSize)) & mask;
508 }
509
510
511 // Convert to array that contains desiredHashSize bits
512 var hashBytes = hash.ToByteArray();
513
514 // Account for missing or extra bytes.
515 if (hashBytes.Length != desiredHashBytes)
516 {
517 var buffer = new byte[desiredHashBytes];
518 Array.Copy(hashBytes, buffer, Math.Min(hashBytes.Length, desiredHashBytes));
519
520 hashBytes = buffer;
521 }
522
523 return new HashValue(hashBytes, desiredHashSize);
524 }
