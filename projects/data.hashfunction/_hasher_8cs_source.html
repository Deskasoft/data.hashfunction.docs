123 public static void Hash(ReadOnlySpan<byte> input, Span<byte> output)
124 {
125 if (output.Length == 32)
126 {
127 fixed (void* ptrOut = output)
128 fixed (void* ptr = input)
129 {
130 var size = input.Length;
131 if (size <= LimitPreemptive)
132 {
133 blake3_hash(ptr, (void*) size, ptrOut);
134 }
135 else
136 {
137 blake3_hash_preemptive(ptr, (void*) size, ptrOut);
138 }
139 }
140 }
141 else
142 {
143 using (var hasher = New())
144 {
145 hasher.Update(input);
146 hasher.Finalize(output);
147 }
148 }
149 }

266 public void Finalize(Span<byte> hash)
267 {
268 if (_hasher == null) ThrowNullReferenceException();
269 ref var pData = ref MemoryMarshal.GetReference(hash);
270 fixed (void* ptr = &pData)
271 {
272 var size = hash.Length;
273 if (size == 32)
274 {
275 blake3_finalize(_hasher, ptr);
276 }
277 else

279 blake3_finalize_xof(_hasher, ptr, (void*)(IntPtr)hash.Length);
