342 private static void Mix(UInt64[] hashValue, ArraySegment<byte> data)
343 {
344 Debug.Assert(data.Count % 96 == 0);
345
346 var dataArray = data.Array;
347 var dataCount = data.Count;
348 var endOffset = data.Offset + dataCount;
349
350 for (var currentOffset = data.Offset; currentOffset < endOffset; currentOffset += 96)
351 {
352 for (var i = 0; i < 12; ++i)
353 {
354 hashValue[i] += BitConverter.ToUInt64(dataArray, currentOffset + (i * 8));
355 hashValue[(i + 2) % 12] ^= hashValue[(i + 10) % 12];
356 hashValue[(i + 11) % 12] ^= hashValue[i];
357 hashValue[i] = RotateLeft(hashValue[i], _MixRotationParameters[i]);
358 hashValue[(i + 11) % 12] += hashValue[(i + 1) % 12];
359 }
360 }
