103 private UInt32 Hash32Len0to4(ArraySegment<byte> data)
104 {
105 var dataArray = data.Array;
106 var dataOffset = data.Offset;
107 var dataCount = data.Count;
108
109 var endOffset = dataOffset + dataCount;
110
111 UInt32 b = 0;
112 UInt32 c = 9;
113
114 for (var currentOffset = dataOffset; currentOffset < endOffset; currentOffset += 1)
115 {
116 b = b * C1 + dataArray[currentOffset];
117 c ^= b;
118 }

120 return Mix(Mur(b, Mur((UInt32) dataCount, c)));

160 private UInt32 Hash32Len25Plus(ArraySegment<byte> data, CancellationToken cancellationToken)
161 {
162
163 var dataArray = data.Array;
164 var dataOffset = data.Offset;
165 var dataCount = data.Count;
166
167 var endOffset = dataOffset + dataCount;
168
169 cancellationToken.ThrowIfCancellationRequested();
170
171 // dataCount > 24
172 UInt32 h = (UInt32) dataCount;
173 UInt32 g = (UInt32) dataCount * C1;
174 UInt32 f = g;
175 {
176 UInt32 a0 = RotateRight(BitConverter.ToUInt32(dataArray, endOffset - 4) * C1, 17) * C2;
177 UInt32 a1 = RotateRight(BitConverter.ToUInt32(dataArray, endOffset - 8) * C1, 17) * C2;
178 UInt32 a2 = RotateRight(BitConverter.ToUInt32(dataArray, endOffset - 16) * C1, 17) * C2;
179 UInt32 a3 = RotateRight(BitConverter.ToUInt32(dataArray, endOffset - 12) * C1, 17) * C2;
180 UInt32 a4 = RotateRight(BitConverter.ToUInt32(dataArray, endOffset - 20) * C1, 17) * C2;
181
182 h ^= a0;
183 h = RotateRight(h, 19);
184 h = h * 5 + 0xe6546b64;
185 h ^= a2;
186 h = RotateRight(h, 19);
187 h = h * 5 + 0xe6546b64;
188
189 g ^= a1;
190 g = RotateRight(g, 19);
191 g = g * 5 + 0xe6546b64;
192 g ^= a3;
193 g = RotateRight(g, 19);
194 g = g * 5 + 0xe6546b64;
195
196 f += a4;
197 f = RotateRight(f, 19);
198 f = f * 5 + 0xe6546b64;
199 }
200
201
202 var groupsToProcess = (dataCount - 1) / 20;
203 var groupEndOffset = dataOffset + (groupsToProcess * 20);
204
205 for (int groupOffset = dataOffset; groupOffset < groupEndOffset; groupOffset += 20)
206 {
207 cancellationToken.ThrowIfCancellationRequested();
208
209 UInt32 a0 = RotateRight(BitConverter.ToUInt32(dataArray, groupOffset + 0) * C1, 17) * C2;
210 UInt32 a1 = BitConverter.ToUInt32(dataArray, groupOffset + 4);
211 UInt32 a2 = RotateRight(BitConverter.ToUInt32(dataArray, groupOffset + 8) * C1, 17) * C2;
212 UInt32 a3 = RotateRight(BitConverter.ToUInt32(dataArray, groupOffset + 12) * C1, 17) * C2;
213 UInt32 a4 = BitConverter.ToUInt32(dataArray, groupOffset + 16);
214
215 h ^= a0;
216 h = RotateRight(h, 18);
217 h = h * 5 + 0xe6546b64;
218
219 f += a1;
220 f = RotateRight(f, 19);
221 f = f * C1;
222
223 g += a2;
224 g = RotateRight(g, 18);
225 g = g * 5 + 0xe6546b64;
226
227 h ^= a3 + a1;
228 h = RotateRight(h, 19);
229 h = h * 5 + 0xe6546b64;
230
231 g ^= a4;
232 g = ReverseByteOrder(g) * 5;
233
234 h += a4 * 5;
235 h = ReverseByteOrder(h);
236
237 f += a0;
238
239 Permute3(ref f, ref h, ref g);
240 }
241
242 cancellationToken.ThrowIfCancellationRequested();
243
244 g = RotateRight(g, 11) * C1;
245 g = RotateRight(g, 17) * C1;
246
247 f = RotateRight(f, 11) * C1;
248 f = RotateRight(f, 17) * C1;
249
250 h = RotateRight(h + g, 19);
251 h = h * 5 + 0xe6546b64;
252 h = RotateRight(h, 17) * C1;
253 h = RotateRight(h + f, 19);
254 h = h * 5 + 0xe6546b64;
255 h = RotateRight(h, 17) * C1;
256
257 return h;
258 }

296 private static UInt64 Hash64Len16(UInt64 u, UInt64 v, UInt64 mul)
297 {
298 UInt64 a = (u ^ v) * mul;
299 a ^= (a >> 47);
300
301 UInt64 b = (v ^ a) * mul;
302 b ^= (b >> 47);
303 b *= mul;
304
305 return b;

308 private UInt64 Hash64Len0to16(ArraySegment<byte> data)
309 {
310 var dataArray = data.Array;
311 var dataOffset = data.Offset;
312 var dataCount = data.Count;
313
314 var endOffset = dataOffset + dataCount;
315
316 if (dataCount >= 8)
317 {
318 UInt64 mul = K2 + (UInt64) dataCount * 2;
319 UInt64 a = BitConverter.ToUInt64(dataArray, dataOffset) + K2;
320 UInt64 b = BitConverter.ToUInt64(dataArray, endOffset - 8);
321 UInt64 c = RotateRight(b, 37) * mul + a;
322 UInt64 d = (RotateRight(a, 25) + b) * mul;
323
324 return Hash64Len16(c, d, mul);
325 }
326
327 if (dataCount >= 4)
328 {
329 UInt64 mul = K2 + (UInt64) dataCount * 2;
330 UInt64 a = BitConverter.ToUInt32(dataArray, dataOffset);
331 return Hash64Len16((UInt64) dataCount + (a << 3), BitConverter.ToUInt32(dataArray, endOffset - 4), mul);
332 }
333
334 if (dataCount > 0)
335 {
336 byte a = dataArray[dataOffset];
337 byte b = dataArray[dataOffset + (dataCount >> 1)];
338 byte c = dataArray[endOffset - 1];
339
340 UInt32 y = (UInt32) a + ((UInt32) b << 8);
341 UInt32 z = (UInt32) dataCount + ((UInt32) c << 2);
342
343 return Mix((UInt64) (y * K2 ^ z * K0)) * K2;
344 }
345
346 return K2;
347 }

401 // Return an 8-byte hash for 33 to 64 bytes.
402 private UInt64 Hash64Len33to64(ArraySegment<byte> data)
403 {
404 var dataArray = data.Array;
405 var dataOffset = data.Offset;
406 var dataCount = data.Count;
407
408 var endOffset = dataOffset + dataCount;
409
410 UInt64 mul = K2 + (UInt64) dataCount * 2;
411 UInt64 a = BitConverter.ToUInt64(dataArray, dataOffset) * K2;
412 UInt64 b = BitConverter.ToUInt64(dataArray, dataOffset + 8);
413 UInt64 c = BitConverter.ToUInt64(dataArray, endOffset - 24);
414 UInt64 d = BitConverter.ToUInt64(dataArray, endOffset - 32);
415 UInt64 e = BitConverter.ToUInt64(dataArray, dataOffset + 16) * K2;
416 UInt64 f = BitConverter.ToUInt64(dataArray, dataOffset + 24) * 9;
417 UInt64 g = BitConverter.ToUInt64(dataArray, endOffset - 8);
418 UInt64 h = BitConverter.ToUInt64(dataArray, endOffset - 16) * mul;
419
420 UInt64 u = RotateRight(a + g,43) + (RotateRight(b, 30) + c) * 9;
421 UInt64 v = ((a + g) ^ d) + f + 1;
422 UInt64 w = ReverseByteOrder((u + v) * mul) + h;
423 UInt64 x = RotateRight(e + f, 42) + c;
424 UInt64 y = (ReverseByteOrder((v + w) * mul) + g) * mul;
425 UInt64 z = e + f + c;
426
427 a = ReverseByteOrder((x + z) * mul + y) + b;
428 b = Mix((z + a) * mul + d + h) * mul;
429 return b + x;
430 }

432 private UInt64 Hash64Len65Plus(ArraySegment<byte> data, CancellationToken cancellationToken)
433 {
434 var dataArray = data.Array;
435 var dataOffset = data.Offset;
436 var dataCount = data.Count;
437
438 var endOffset = dataOffset + dataCount;
439
440 // For strings over 64 bytes we hash the end first, and then as we
441 // loop we keep 56 bytes of state: v, w, x, y, and z.
442 UInt64 x = BitConverter.ToUInt64(dataArray, endOffset - 40);
443 UInt64 y = BitConverter.ToUInt64(dataArray, endOffset - 16) + BitConverter.ToUInt64(dataArray, endOffset - 56);
444 UInt64 z = Hash64Len16(
445 BitConverter.ToUInt64(dataArray, endOffset - 48) + (UInt64) dataCount,
446 BitConverter.ToUInt64(dataArray, endOffset - 24));
447
448 UInt128 v = WeakHashLen32WithSeeds(dataArray, endOffset - 64, (UInt64) dataCount, z);
449 UInt128 w = WeakHashLen32WithSeeds(dataArray, endOffset - 32, y + K1, x);
450
451 x = x * K1 + BitConverter.ToUInt64(dataArray, 0);
452
453 // For each 64-byte chunk
454 var groupEndOffset = dataOffset + (dataCount - (dataCount % 64));
455
456 for (var currentOffset = dataOffset; currentOffset < groupEndOffset; currentOffset += 64)
457 {
458 cancellationToken.ThrowIfCancellationRequested();
459
460 x = RotateRight(x + y + v.Low + BitConverter.ToUInt64(dataArray, currentOffset + 8), 37) * K1;
461 y = RotateRight(y + v.High + BitConverter.ToUInt64(dataArray, currentOffset + 48), 42) * K1;
462 x ^= w.High;
463 y += v.Low + BitConverter.ToUInt64(dataArray, currentOffset + 40);
464 z = RotateRight(z + w.Low, 33) * K1;
465 v = WeakHashLen32WithSeeds(dataArray, currentOffset, v.High * K1, x + w.Low);
466 w = WeakHashLen32WithSeeds(dataArray, currentOffset + 32, z + w.High, y + BitConverter.ToUInt64(dataArray, currentOffset + 16));
467
468 UInt64 temp = x;
469 x = z;
470 z = temp;
471 }
472
473 return Hash64Len16(Hash64Len16(v.Low, w.Low) + Mix(y) * K1 + z,
474 Hash64Len16(v.High, w.High) + x);
475 }

511 private UInt128 CityHash128WithSeed(ArraySegment<byte> data, UInt128 seed, CancellationToken cancellationToken)
512 {
513 cancellationToken.ThrowIfCancellationRequested();
514
515 var dataCount = data.Count;
516
517 if (dataCount < 128)
518 return CityMurmur(data, seed);
519
520
521 var dataArray = data.Array;
522 var dataOffset = data.Offset;
523
524 var endOffset = dataOffset + dataCount;
525
526 // We expect len >= 128 to be the common case. Keep 56 bytes of state:
527 // v, w, x, y, and z.
528 UInt128 v;
529 {
530 var vLow = RotateRight(seed.High ^ K1, 49) * K1 + BitConverter.ToUInt64(dataArray, dataOffset);
531 v = new UInt128(
532 vLow,
533 RotateRight(vLow, 42) * K1 + BitConverter.ToUInt64(dataArray, dataOffset + 8));
534 }
535
536
537 UInt128 w = new UInt128(
538 RotateRight(seed.High + ((UInt64) dataCount * K1), 35) * K1 + seed.Low,
539 RotateRight(seed.Low + BitConverter.ToUInt64(dataArray, dataOffset + 88), 53) * K1);
540
541 UInt64 x = seed.Low;
542 UInt64 y = seed.High;
543 UInt64 z = (UInt64) dataCount * K1;
544
545
546
547 // This is the same inner loop as CityHash64()
548 int lastGroupEndOffset;
549 {
550 var groupEndOffset = dataOffset + (dataCount - (dataCount % 128));
551
552 for (var groupCurrentOffset = dataOffset; groupCurrentOffset < groupEndOffset; groupCurrentOffset += 128)
553 {
554 cancellationToken.ThrowIfCancellationRequested();
555
556 x = RotateRight(x + y + v.Low + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 8), 37) * K1;
557 y = RotateRight(y + v.High + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 48), 42) * K1;
558 x ^= w.High;
559 y += v.Low + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 40);
560 z = RotateRight(z + w.Low, 33) * K1;
561 v = WeakHashLen32WithSeeds(dataArray, groupCurrentOffset, v.High * K1, x + w.Low);
562 w = WeakHashLen32WithSeeds(dataArray, groupCurrentOffset + 32, z + w.High, y + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 16));
563
564 {
565 UInt64 temp = z;
566 z = x;
567 x = temp;
568 }
569
570 x = RotateRight(x + y + v.Low + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 72), 37) * K1;
571 y = RotateRight(y + v.High + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 112), 42) * K1;
572 x ^= w.High;
573 y += v.Low + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 104);
574 z = RotateRight(z + w.Low, 33) * K1;
575 v = WeakHashLen32WithSeeds(dataArray, groupCurrentOffset + 64, v.High * K1, x + w.Low);
576 w = WeakHashLen32WithSeeds(dataArray, groupCurrentOffset + 96, z + w.High, y + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 80));
577
578 {
579 UInt64 temp = z;
580 z = x;
581 x = temp;
582 }
583 }
584
585 lastGroupEndOffset = groupEndOffset;
586 }
587
588 cancellationToken.ThrowIfCancellationRequested();
589
590 x += RotateRight(v.Low + z, 49) * K0;
591 y = y * K0 + RotateRight(w.High, 37);
592 z = z * K0 + RotateRight(w.Low, 27);
593 w = new UInt128(w.Low * 9, w.High);
594 v = new UInt128(v.Low * K0, v.High);
595
596
597
598 // Hash up to 4 chunks of 32 bytes each from the end of data.
599 {
600 var groupEndOffset = lastGroupEndOffset - 32;
601
602 for (var groupCurrentOffset = endOffset - 32; groupCurrentOffset > groupEndOffset; groupCurrentOffset -= 32)
603 {
604 cancellationToken.ThrowIfCancellationRequested();
605
606 y = RotateRight(x + y, 42) * K0 + v.High;
607 w = new UInt128(w.Low + BitConverter.ToUInt64(dataArray, groupCurrentOffset + 16), w.High);
608 x = x * K0 + w.Low;
609 z += w.High + BitConverter.ToUInt64(dataArray, groupCurrentOffset);
610 w = new UInt128(w.Low, w.High + v.Low);
611 v = WeakHashLen32WithSeeds(dataArray, groupCurrentOffset, v.Low + z, v.High);
612 v = new UInt128(v.Low * K0, v.High);
613 }
614 }
615
616 // At this point our 56 bytes of state should contain more than
617 // enough information for a strong 128-bit hash. We use two
618 // different 56-byte-to-8-byte hashes to get a 16-byte final result.
619 x = Hash64Len16(x, v.Low);
620 y = Hash64Len16(y + z, w.Low);
621
622 return new UInt128(
623 Hash64Len16(x + v.High, w.High) + y,
624 Hash64Len16(x + w.High, y + v.High));
625 }

629 // A subroutine for CityHash128(). Returns a decent 128-bit hash for strings
630 // of any length representable in signed long. Based on City and Murmur.
631 private UInt128 CityMurmur(ArraySegment<byte> data, UInt128 seed)
632 {
633 var dataArray = data.Array;
634 var dataOffset = data.Offset;
635 var dataCount = data.Count;
636
637 var endOffset = dataOffset + dataCount;
638
639 UInt64 a = seed.Low;
640 UInt64 b = seed.High;
641 UInt64 c;
642 UInt64 d;
643
644 if (dataCount <= 16)
645 {
646 // len <= 16
647 a = Mix(a * K1) * K1;
648 c = b * K1 + Hash64Len0to16(data);
649 d = Mix(a + (dataCount >= 8 ? BitConverter.ToUInt64(dataArray, dataOffset) : c));
650
651 } else {
652 // len > 16
653 c = Hash64Len16(BitConverter.ToUInt64(dataArray, endOffset - 8) + K1, a);
654 d = Hash64Len16(b + (UInt64) dataCount, c + BitConverter.ToUInt64(dataArray, endOffset - 16));
655 a += d;
656
657 var groupEndOffset = dataOffset + dataCount - 16;
658
659 for (var groupCurrentOffset = dataOffset; groupCurrentOffset < groupEndOffset; groupCurrentOffset += 16)
660 {
661 a ^= Mix(BitConverter.ToUInt64(dataArray, groupCurrentOffset) * K1) * K1;
662 a *= K1;
663 b ^= a;
664 c ^= Mix(BitConverter.ToUInt64(dataArray, groupCurrentOffset + 8) * K1) * K1;
665 c *= K1;
666 d ^= c;
667 }
668 }
669
670 a = Hash64Len16(a, c);
671 b = Hash64Len16(d, b);
672 return new UInt128(a ^ b, Hash64Len16(b, a));
673 }

679 private static UInt32 Mix(UInt32 h)
