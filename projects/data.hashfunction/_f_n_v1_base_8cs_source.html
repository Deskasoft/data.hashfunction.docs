167 protected static UInt32[] ExtendedMultiply(IReadOnlyList<UInt32> operand1, IReadOnlyList<UInt32> operand2, int hashSizeInBytes)
168 {
169 Debug.Assert(hashSizeInBytes % 4 == 0);
170
171 // Temporary array to hold the results of 32-bit multiplication.
172 var product = new UInt32[hashSizeInBytes / 4];
173
174 // Bottom of equation
175 for (int y = 0; y < operand2.Count; ++y)
176 {
177 // Skip multiplying things by zero
178 if (operand2[y] == 0)
179 continue;
180
181 UInt32 carryOver = 0;
182
183 // Top of equation
184 for (int x = 0; x < operand1.Count; ++x)
185 {
186 if (x + y >= product.Length)
187 break;
188
189 var productResult = product[x + y] + (((UInt64)operand2[y]) * operand1[x]) + carryOver;
190 product[x + y] = (UInt32)productResult;
191
192 carryOver = (UInt32)(productResult >> 32);
193 }
194 }
195
196 return product;
197 }
