153 {
154 UInt32 hashValue;
155 {
156 if (_bytesProcessed > 0)
157 hashValue = RotateLeft(_a, 1) + RotateLeft(_b, 7) + RotateLeft(_c, 12) + RotateLeft(_d, 18);

169 // In 4-byte chunks, process all process all full chunks
