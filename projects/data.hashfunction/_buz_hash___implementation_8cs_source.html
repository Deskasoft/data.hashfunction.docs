30 public BuzHash_Implementation(IBuzHashConfig config)
31 {
32 if (config == null)
33 throw new ArgumentNullException(nameof(config));
34
35 _config = config.Clone();
36
37
38 if (_config.Rtab == null || _config.Rtab.Count != 256)
39 throw new ArgumentException($"{nameof(config.Rtab)} must be non-null list of 256 {nameof(UInt64)} values.", $"{nameof(config)}.{nameof(config.Rtab)}");
40
41 if (!_validHashSizes.Contains(_config.HashSizeInBits))
42 throw new ArgumentOutOfRangeException($"{nameof(config)}.{nameof(config.HashSizeInBits)}", _config.HashSizeInBits, $"{nameof(config)}.{nameof(config.HashSizeInBits)} must be contained within BuzHashBase.ValidHashSizes.");
43
44 if (_config.ShiftDirection != CircularShiftDirection.Left && _config.ShiftDirection != CircularShiftDirection.Right)
45 throw new ArgumentOutOfRangeException($"{nameof(config)}.{nameof(config.ShiftDirection)}", _config.ShiftDirection, $"{nameof(config)}.{nameof(config.ShiftDirection)} must be a valid {nameof(CircularShiftDirection)} value.");
46 }

102 protected override void TransformByteGroupsInternal(ArraySegment<byte> data)
103 {
104 var dataArray = data.Array;
105 var dataCount = data.Count;
106 var endOffset = data.Offset + dataCount;
107
108 var tempHashValue = _hashValue;
109 var tempRtab = _rtab;
110
111 for (var currentOffset = data.Offset; currentOffset < endOffset; ++currentOffset)
112 tempHashValue = (byte) (CShift(tempHashValue, 1, _shiftDirection) ^ (byte) tempRtab[dataArray[currentOffset]]);
113
114 _hashValue = tempHashValue;
115 }
116
117 protected override IHashValue FinalizeHashValueInternal(CancellationToken cancellationToken) =>
118 new HashValue(new byte[] { _hashValue }, 8);
